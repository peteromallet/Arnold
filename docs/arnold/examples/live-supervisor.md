class IssueDifficulty(IntEnum):
    """1-5 difficulty scale, matching the partnered-5 execute tier model fan-out."""

    TRIVIAL = 1
    SIMPLE = 2
    MODERATE = 3
    HARD = 4
    SEVERE = 5

def _assess_difficulty(incident: Incident, category: HealthCategory) -> IssueDifficulty:
    """Score issue intricacy 1-5 so the repair can fan out to the right model."""
    findings = incident.signals.doctor_findings
    finding_checks = {f.check for f in findings}
    harness_checks = finding_checks & {"stale_lock", "orphan_subprocess", "multiple_checkouts"}

    # 1: trivial harness cleanup (single mechanical issue).
    if category == HealthCategory.HARNESS_ISSUE and len(harness_checks) == 1:
        return IssueDifficulty.TRIVIAL

    # 2: multiple harness cleanups, or a blocked plan with a clear recovery path.
    if category == HealthCategory.HARNESS_ISSUE:
        return IssueDifficulty.SIMPLE
    if category == HealthCategory.PLAN_ISSUE:
        block = incident.signals.block_details
        if block.get("recoverable_via") and len(finding_checks) <= 1:
            return IssueDifficulty.SIMPLE

    # 3: complex blocked plan, false stall, or a recently dead plan.
    if category == HealthCategory.PLAN_ISSUE:
        return IssueDifficulty.MODERATE
    if category == HealthCategory.FALSE_STALL:
        return IssueDifficulty.MODERATE
    if category == HealthCategory.DEAD_OR_DISAPPEARED:
        age = incident.signals.last_event_age_seconds
        if age is None or age <= 3600:
            return IssueDifficulty.MODERATE
        if age <= 86400:
            return IssueDifficulty.HARD
        return IssueDifficulty.SEVERE

    # 4: environment/repo-level issues.
    if category == HealthCategory.ENVIRONMENT_ISSUE:
        return IssueDifficulty.HARD

    # 5: unknown/degraded signals need the strongest model.
    return IssueDifficulty.SEVERE

def _select_model_for_difficulty(difficulty: IssueDifficulty) -> str:
    """Map difficulty 1-5 to a repair model, mirroring partnered-5 execute tier_models."""
    return _EXECUTE_TIER_MODELS.get(int(difficulty), _EXECUTE_TIER_MODELS[5])

def _diagnosis_reasoning(incident: Incident, category: HealthCategory) -> str:
    signals = incident.signals
    parts = [
        f"liveness={signals.liveness}",
        f"triage={incident.triage.value}",
        f"category={category.value}",
    ]
    if signals.has_in_flight_llm:
        parts.append("has_in_flight_llm=True")
    if signals.last_event_age_seconds is not None:
        parts.append(f"last_event_age_seconds={signals.last_event_age_seconds:.1f}")
    if signals.block_details:
        parts.append(f"block_details={signals.block_details}")
    return "; ".join(parts)

def _has_stale_lock(findings: tuple[Any, ...]) -> bool:
    return any(f.check == "stale_lock" for f in findings)

def _load_spec(path: Path) -> dict[str, Any] | None:
    try:
        text = path.read_text(encoding="utf-8")
    except Exception:
        return None
    try:
        return json.loads(text)

    try:

        return data if isinstance(data, dict) else None
    except Exception:

def _has_pending_milestones(chain_spec_path: str | None) -> bool:
    """True if the chain spec has milestones that are not all completed."""
    if not chain_spec_path:
        return False
    spec_path = Path(chain_spec_path)
    if not spec_path.is_file():
        return False
    spec = _load_spec(spec_path)
    if spec is None:
        return False
    milestones = spec.get("milestones")
    if not isinstance(milestones, list) or not milestones:
        return False
    state_path = spec_path.with_name("chain_state.json")
    completed: set[str] = set()
    if state_path.is_file():
        try:
            state = json.loads(state_path.read_text(encoding="utf-8"))
            completed = set(state.get("completed_milestones") or ())
        except Exception:
            pass
    return any(m.get("label") not in completed for m in milestones)

def _build_allowlist_context(incident: Incident) -> dict[str, Any]:
    signals = incident.signals
    orphan_pids = _extract_orphan_pids(signals.doctor_findings)
    return {
        "plan_name": incident.plan_entry.plan_name,
        "plan_dir": incident.plan_entry.plan_dir,
        "state": incident.plan_entry.state,
        "block_details": signals.block_details,
        "is_resumable": bool(signals.block_details.get("recoverable_via")),
        "chain_spec_path": incident.plan_entry.chain_spec_path,
        "has_pending_milestones": _has_pending_milestones(incident.plan_entry.chain_spec_path),
        "orphan_pids": orphan_pids,
    }

def _is_terminal_state(state: dict[str, Any]) -> bool:
    terminal = {
        "completed",
        "failed",
        "aborted",
        "resolved",
        "cancelled",
        "finalized",
        "executed",
        "done",
        "reviewed",
        "accepted",
    }
    current = state.get("current_state")
    if isinstance(current, str) and current in terminal:
        return True
    return state.get("status") in terminal

def _select_repair_recommendation(
    incident: Incident,
    category: HealthCategory,
    context: dict[str, Any],
) -> RepairRecommendation:
    """Choose a status-aware repair recommendation for an incident."""
    difficulty = _assess_difficulty(incident, category)
    model = _select_model_for_difficulty(difficulty)
    base_context = {
        "plan_name": incident.plan_entry.plan_name,
        "plan_dir": incident.plan_entry.plan_dir,
        "difficulty": difficulty.value,
        "model": model,
    }

    # 1. Orphan subprocesses are the highest-priority cleanup.
    orphan_pids = context.get("orphan_pids") or []
    if orphan_pids:
        first_pid = orphan_pids[0]
        target_category = "unknown"
        for finding in incident.signals.doctor_findings:
            if finding.check == "orphan_subprocess" and f"pid={first_pid}" in finding.message:
                parts = finding.message.split()
                if len(parts) >= 2:
                    target_category = parts[1]
                break
        return RepairRecommendation(
            command=f"kill -9 {first_pid}",
            context={**base_context, "orphan_pids": orphan_pids, "target_category": target_category},
        )

    state = incident.plan_entry.state or {}

    findings = incident.signals.doctor_findings

    # 2. Terminal plans with stale locks just need the lock cleaned up.
    if _is_terminal_state(state) and _has_stale_lock(findings):
        lock_path = Path(incident.plan_entry.plan_dir) / ".plan.lock"
        return RepairRecommendation(
            command=f"rm {lock_path}",
            context={**base_context, "reason": "terminal plan with stale lock"},
        )

    # 2b. Non-terminal blocked plans with stale locks need to be resumed; the
    # harness finding is real, but lock cleanup alone will not unblock the plan.
    if category == HealthCategory.HARNESS_ISSUE and _has_stale_lock(findings):
        if context.get("is_resumable"):
            return RepairRecommendation(
                command="auto",
                context={**base_context, "state": state, "block_details": context.get("block_details", {})},
            )
        if context.get("block_details", {}).get("recoverable_via"):
            return RepairRecommendation(
                command="resume",
                context={**base_context, "is_resumable": True},
            )

    # 3. Active blocked/resumable plans should auto-resume.
    if category in (HealthCategory.PLAN_ISSUE, HealthCategory.FALSE_STALL):
        if context.get("is_resumable"):
            return RepairRecommendation(
                command="auto",
                context={**base_context, "state": state, "block_details": context.get("block_details", {})},
            )
        if context.get("block_details", {}).get("recoverable_via"):
            return RepairRecommendation(
                command="resume",
                context={**base_context, "is_resumable": True},
            )

    # 4. Plans with a chain spec and pending milestones but no live process.
    if (
        incident.triage in (Triage.STALE, Triage.MAYBE_LIVE)
        and context.get("chain_spec_path")
        and context.get("has_pending_milestones")
    ):
        return RepairRecommendation(
            command="chain start --one --no-git-refresh --no-push",
            context={**base_context, "chain_spec_path": context["chain_spec_path"]},
        )

    # 5. Safe default: inspect.
    return RepairRecommendation(command="doctor", context=base_context)

class RecheckEmitStep:

            "resumable": True,

        _write_json(path, payload)

        return StepResult(
            outputs={"recheck_emit": str(path)},
            state_patch={"recheck_emit": payload},
            next="halt",
        )

````markdown
# live-supervisor

Input: a Snapshot dict passed as `initial_state={"snapshot": <dict>}`. The
snapshot contains a `scan_ts_utc` ISO timestamp and a list of `incidents`, each
with a `plan_entry` and a pre-computed `signal_bundle`.

Output: four JSON artifact files written under the supplied
`RuntimeEnvelope.artifact_root`:

- `classify/classifications.json` — per-incident health category
- `diagnose/diagnoses.json` — per-incident diagnosis and normalized findings
- `repair_decision/repair_decisions.json` — recommended action + allowlist verdict
- `recheck_emit/recheck_emit.json` — `recheck_after` timestamp and resumable flag

The pipeline never sleeps, shells out, or makes network requests. Repair-agent
credentials may be absent; in that case the repair decision degrades to
`no_repair_available` rather than failing.
````
