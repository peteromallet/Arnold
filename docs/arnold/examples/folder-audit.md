def _build_tree(
    path: Path,
    max_depth: int = 3,
    respect_gitignore: bool = True,
    skip_hidden: bool = True,
    _current_depth: int = 0,
    _relative_root: Path | None = None,
) -> list[dict[str, Any]]:
    """Walk *path* and return a flat-ish list of directory-entry dicts.

    Returns a list where each directory (including the root) is an entry
    dict with ``path`` (relative string), ``level`` (int depth), and
    ``children`` (list of child name/type dicts).  The root entry
    (``path="."``) is always first.

    When *respect_gitignore* is True, patterns from ``.gitignore`` are
    applied. When *skip_hidden* is True, entries whose name starts with
    ``.`` are excluded.
    """
    if _current_depth > max_depth or not path.is_dir():
        return []

    if _relative_root is None:
        _relative_root = path

    # Load gitignore patterns if requested
    gitignore_patterns: set[str] = set()
    if respect_gitignore:
        gitignore_path = path / ".gitignore"
        if gitignore_path.is_file():
            gitignore_patterns = set(
                line.strip()
                for line in gitignore_path.read_text(encoding="utf-8").splitlines()
                if line.strip() and not line.strip().startswith("#")
            )

    children: list[dict[str, Any]] = []
    sub_results: list[dict[str, Any]] = []

    try:
        for child in sorted(path.iterdir()):
            cname = child.name

            # Skip hidden
            if skip_hidden and cname.startswith("."):
                continue

            # Skip gitignore'd
            if cname in gitignore_patterns or (
                child.is_dir() and (cname + "/") in gitignore_patterns
            ):
                continue

            is_dir = child.is_dir()
            child_entry: dict[str, Any] = {
                "name": cname,
                "type": "dir" if is_dir else "file",
            }
            children.append(child_entry)

            if is_dir and _current_depth < max_depth:
                sub_results.extend(
                    _build_tree(
                        child,
                        max_depth=max_depth,
                        respect_gitignore=respect_gitignore,
                        skip_hidden=skip_hidden,
                        _current_depth=_current_depth + 1,
                        _relative_root=_relative_root,
                    )
                )
    except (OSError, PermissionError):
        pass

    # Compute relative path for this directory
    try:
        rel = str(path.relative_to(_relative_root))
    except ValueError:
        rel = str(path)
    if rel == "." or rel == "":
        rel = "."

    current_entry: dict[str, Any] = {
        "path": rel,
        "level": _current_depth,
        "children": children,
    }

    # Root first, then subdirectories
    if _current_depth == 0:
        return [current_entry] + sub_results
    else:
        return [current_entry] + sub_results

def _summarize_children(
    children: list[dict[str, Any]], max_files: int = 50
) -> list[dict[str, Any]]:
    """Cap the number of file entries in *children* to *max_files*.

    Directory entries are always included. When files exceed *max_files*,
    only the first *max_files* files are kept and a placeholder entry
    ``{"name": "... (N more files)"}`` is appended.
    """
    dirs = [c for c in children if c.get("type") == "dir"]
    files = [c for c in children if c.get("type") != "dir"]

    if len(files) <= max_files:
        return dirs + files

    kept = files[:max_files]
    omitted = len(files) - max_files
    placeholder: dict[str, Any] = {"name": f"... ({omitted} more files)"}
    return dirs + kept + [placeholder]

def _compute_summary(folders: list[dict[str, Any]]) -> dict[str, Any]:
    """Compute aggregate summary counts from classified *folders*.

    Each folder dict is expected to have an ``items`` list where each
    item has a ``classification`` string field.
    """
    total_folders = len(folders)
    total_items = 0
    counts: dict[str, int] = {}

    for folder in folders:
        for item in folder.get("items", []):
            total_items += 1
            cls = item.get("classification", "unknown")
            counts[cls] = counts.get(cls, 0) + 1

    result: dict[str, Any] = {
        "total_folders": total_folders,
        "total_items": total_items,
    }
    result.update(counts)
    return result
