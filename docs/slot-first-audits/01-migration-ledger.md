-- M1 verification query: compare this ledger with live dependencies.
WITH legacy_objects AS (
  SELECT c.oid, n.nspname, c.relname AS name, c.relkind::text AS kind
  FROM pg_class c
  JOIN pg_namespace n ON n.oid = c.relnamespace
  WHERE n.nspname = 'public'
    AND c.relname IN ('generations', 'generation_variants', 'shot_generations')
), dependents AS (
  SELECT
    lo.name AS legacy_object,
    d.classid::regclass::text AS dependent_catalog,
    d.objid,
    d.refobjid,
    d.deptype,
    pg_describe_object(d.classid, d.objid, d.objsubid) AS dependent_object,
    pg_describe_object(d.refclassid, d.refobjid, d.refobjsubid) AS referenced_object
  FROM legacy_objects lo
  JOIN pg_depend d ON d.refobjid = lo.oid
)
SELECT *
FROM dependents
ORDER BY legacy_object, dependent_object;
